//// -- LEVEL 1
//// -- Tables and References

// Creating tables
Table Animals {
    Id              INT      [pk]
    Name            NVARCHAR 
    Species         NVARCHAR 
    Gender          INT         
    Age             FLOAT 
    Weight          FLOAT 
    Height         FLOAT 
    ClimatZone     INT       
    DietType        INT         
    EnclosureNumber INT     
    DateJoined      DATETIME2 
    NeedsRelocation BIT    
    MedicalRecordId INT       
    MenuId         INT    
    AnimalId       INT        
    WorkerId       INT      
}

Ref: Animals.MedicalRecordId > MedicalRecords.Id
Ref: Animals.MenuId > Menus.Id
Ref: Animals.WorkerId > Workers.Id
  
Table Deliveries {
    Id       INT    
    ProductId  INT   
    Amount    FLOAT 
    Price     DECIMAL 
    Date       DATETIME2
    SupplierId INT   
}

Ref: Deliveries.SupplierId > Suppliers.Id
Ref: Deliveries.ProductId > Foods.Id

Table Foods {
    Id        INT [pk, increment]
    Type       INT
    SupplierId INT 
}

Ref: Foods.SupplierId > Suppliers.Id

Table Illnesses {
    Id                INT  [pk]       
    Name              INT       
    IsolationRequired BIT          
    DateStarted       DATETIME2 
    DateEnded         DATETIME2
    MedicalRecordId   INT         
}

Ref: Illnesses.MedicalRecordId > MedicalRecords.Id

Table Meals {
    Id     INT  [pk]
    FoodId INT  
    Amount FLOAT 
    Time   TIME 
    MenuId INT     
}

Ref: Meals.FoodId > Foods.Id
Ref: Meals.MenuId > Menus.Id

Table MedicalRecords {
    Id     INT  [pk]       
    Fertility      BIT    
    ExpectingChildren   BIT    
    DateExpectingChildren DATETIME2 
}

Table Menus {
     Id   INT  [pk]
}

Table Suppliers {
   Id        INT [pk, increment]
   Name      NVARCHAR 
   DateJoined DATETIME2 
}

Table Trades {
    Id  INT [pk, increment]
    ZooName   NVARCHAR 
    Date      DATETIME2 
    TradeType INT         
    AnimalId  INT      
}

Ref: Trades.AnimalId > Animals.Id

Table Vaccinations {
    Id   INT  [pk]
    Vaccine   INT     
    Date    DATETIME2 
    MedicalRecordId INT   
}

Ref: Vaccinations.MedicalRecordId > MedicalRecords.Id

Table Workers {
    Id   INT  [pk]
    FirstName  NVARCHAR
    LastName   NVARCHAR 
    Gender     INT    
    Age        REAL        
    Occupation INT          
    Payrate    DECIMAL 
    DateJoined DATETIME2 
}
